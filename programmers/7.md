import java.util.*;

class Solution {
    public int solution(int n) {
        int answer = 0;
        boolean loop_n = true;
        ArrayList<Integer> new_n = new ArrayList<Integer>();
        while(loop_n){
            new_n.add(n%3);
            n = n/3;
            if(n<1){
                loop_n = false;
            }
        }
        int num = new_n.size()-1;
        for(int i: new_n){
            System.out.println(i);
            answer += Math.pow(3,num) * i;
            num -= 1;
        }
        return answer;
    }
}
